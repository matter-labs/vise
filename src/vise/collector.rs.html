12type CollectorFn<M> = Box<dyn Fn() -> M + Send + Sync>;

14/// Error that can occur when calling [`Collector::before_scrape()`].
15#[derive(Debug)]
16pub struct BeforeScrapeError(());
17
18impl fmt::Display for BeforeScrapeError {
19    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
20        formatter.write_str("Cannot set collector function: it is already set")
21    }
22}
23
24impl error::Error for BeforeScrapeError {}

26/// Collector allowing to define metrics dynamically.
27///
28/// In essence, a collector is a lazily initialized closure producing [`Metrics`] which is called
29/// each time a [`Registry`] it's registered with is being scraped.
30///
31/// ## Sharing state
32///
33/// Because of lazy initialization, the collector closure has access to the shared state with the rest
34/// of the app. **Beware that `Collector`s live indefinitely.** To avoid resource leaks, use [`Weak`]
35/// or similar types that do not lengthen the lifetime of the app state. Because `Metric` is
36/// implemented for `Option`s, if the tracked state is dropped, you can simply return `None`
37/// from the closure.
38///
39/// [`Weak`]: std::sync::Weak
40pub struct Collector<M> {
41    inner: OnceCell<CollectorFn<M>>,
42}
43
44impl<M> fmt::Debug for Collector<M> {
45    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
46        formatter
47            .debug_struct("Collector")
48            .field("inner", &self.inner.get().map(|_| "_"))
49            .finish()
50    }
51}
52
53impl<M: Metrics> Default for Collector<M> {
54    fn default() -> Self {
55        Self::new()
56    }
57}
58
59impl<M: Metrics> Collector<M> {
60    /// Creates a new collector.
61    pub const fn new() -> Self {
62        Self {
63            inner: OnceCell::new(),
64        }
65    }
66
67    /// Initializes the producing function for this collector. The function will be called each time
68    /// a [`Registry`] the collector is registered in is scraped.
69    ///
70    /// # Errors
71    ///
72    /// Returns an error if the producing function has been already set.
73    pub fn before_scrape<F>(&'static self, hook: F) -> Result<(), BeforeScrapeError>
74    where
75        F: Fn() -> M + 'static + Send + Sync,
76    {
77        self.inner
78            .set(Box::new(hook))
79            .map_err(|_| BeforeScrapeError(()))
80    }
81}
82
83impl<M: Metrics> CollectorTrait for &'static Collector<M> {
84    fn encode(&self, encoder: DescriptorEncoder<'_>) -> fmt::Result {
85        if let Some(hook) = self.inner.get() {
86            let mut visitor = MetricsEncoder::from(encoder);
87            hook().visit_metrics(&mut visitor);
88            visitor.check()
89        } else {
90            Ok(())
91        }
92    }
93}
94
95impl<M: Metrics> CollectToRegistry for Collector<M> {
96    fn descriptor(&self) -> &'static MetricGroupDescriptor {
97        &M::DESCRIPTOR
98    }
99
100    fn collect_to_registry(&'static self, registry: &mut Registry) {
101        registry.register_collector(self);
102    }
103}

105/// Lazy collector of `Global` metrics. Only exports metrics once they are initialized;
106/// does not initialize metrics on its own.
107pub(crate) struct LazyGlobalCollector<M: Metrics>(&'static Lazy<M>);

109impl<M: Metrics> fmt::Debug for LazyGlobalCollector<M> {

111        formatter
112            .debug_struct("LazyGlobalCollector")
113            .finish_non_exhaustive()
114    }
115}

117impl<M: Metrics> LazyGlobalCollector<M> {
118    pub(crate) fn new(metrics: &'static Lazy<M>) -> Self {
119        Self(metrics)
120    }
121}

123impl<M: Metrics> CollectorTrait for LazyGlobalCollector<M> {
124    fn encode(&self, encoder: DescriptorEncoder<'_>) -> fmt::Result {
125        if let Some(metrics) = Lazy::get(self.0) {
126            let mut visitor = MetricsEncoder::from(encoder);
127            metrics.visit_metrics(&mut visitor);
128            visitor.check()
129        } else {
130            Ok(())
131        }
132    }

135#[cfg(test)]
136mod tests {
137    use std::sync::{
138        atomic::{AtomicI64, Ordering},
139        Arc,
140    };
141
142    use once_cell::sync::Lazy;
143
144    use super::*;
145    use crate::{Format, Gauge, Registry, Unit};
146
147    #[derive(Debug, Metrics)]
148    #[metrics(crate = crate, prefix = "dynamic")]
149    struct TestMetrics {
150        /// Test gauge.
151        #[metrics(unit = Unit::Bytes)]
152        gauge: Gauge,
153    }
154
155    /// Collector that produces owned metrics (useful if resource consumption is a concern).
156    /// Metrics are also `Option`al to account for a potentially dropped data source.
157    #[crate::register]
158    #[metrics(crate = crate)]
159    static OWNING_COLLECTOR: Collector<Option<TestMetrics>> = Collector::new();
160
161    #[test]
162    fn using_owning_collector() {
163        let state = Arc::new(AtomicI64::new(0));
164        let state_for_collector = Arc::downgrade(&state);
165
166        OWNING_COLLECTOR
167            .before_scrape(move || {
168                let state = state_for_collector.upgrade()?;
169                let metrics = TestMetrics::default();
170                metrics.gauge.set(state.load(Ordering::Relaxed));
171                Some(metrics)
172            })
173            .unwrap();
174
175        let mut registry = Registry::empty();
176        registry.register_collector(&OWNING_COLLECTOR);
177        assert_collector_works(&registry, &state);
178
179        drop(state);
180        let mut buffer = String::new();
181        registry.encode(&mut buffer, Format::OpenMetrics).unwrap();
182
183        assert_eq!(buffer, "# EOF\n");
184    }
185
186    fn assert_collector_works(registry: &Registry, state: &Arc<AtomicI64>) {
187        state.store(123, Ordering::Release);
188        let mut buffer = String::new();
189        registry.encode(&mut buffer, Format::OpenMetrics).unwrap();
190        let lines: Vec<_> = buffer.lines().collect();
191
192        let expected_lines = [
193            "# HELP dynamic_gauge_bytes Test gauge.",
194            "# TYPE dynamic_gauge_bytes gauge",
195            "# UNIT dynamic_gauge_bytes bytes",
196            "dynamic_gauge_bytes 123",
197        ];
198        for line in expected_lines {
199            assert!(expected_lines.contains(&line), "{lines:#?}");
200        }
201    }
202
203    /// Collector that produces borrowed metrics (useful if we want to update *some* of the metrics
204    /// outside scraping / conditionally).
205    static BORROWING_COLLECTOR: Collector<&'static TestMetrics> = Collector::new();
206
207    /// Source of the collector.
208    static METRICS_INSTANCE: Lazy<TestMetrics> = Lazy::new(TestMetrics::default);
209
210    #[test]
211    fn using_borrowing_collector() {
212        let state = Arc::new(AtomicI64::new(0));
213        let state_for_collector = Arc::downgrade(&state);
214
215        BORROWING_COLLECTOR
216            .before_scrape(move || {
217                let metrics = &METRICS_INSTANCE;
218                if let Some(state) = state_for_collector.upgrade() {
219                    metrics.gauge.set(state.load(Ordering::Relaxed));
220                }
221                metrics
222            })
223            .unwrap();
224
225        let mut registry = Registry::empty();
226        registry.register_collector(&BORROWING_COLLECTOR);
227        assert_collector_works(&registry, &state);
228
229        METRICS_INSTANCE.gauge.set(42);
230        drop(state);
231
232        let mut buffer = String::new();
233        registry.encode(&mut buffer, Format::OpenMetrics).unwrap();
234        let lines: Vec<_> = buffer.lines().collect();
235        assert!(lines.contains(&"dynamic_gauge_bytes 42"), "{lines:#?}");
236    }
