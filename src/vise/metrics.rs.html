16/// Collection of metrics for a library or application. Should be derived using the corresponding macro.
17pub trait Metrics: 'static + Send + Sync {
18    /// Metrics descriptor.
19    const DESCRIPTOR: MetricGroupDescriptor;
20
21    #[doc(hidden)] // implementation detail
22    fn visit_metrics(&self, visitor: &mut dyn MetricsVisitor);
23}
24
25impl<M: Metrics> Metrics for &'static M {
26    const DESCRIPTOR: MetricGroupDescriptor = M::DESCRIPTOR;
27
28    fn visit_metrics(&self, visitor: &mut dyn MetricsVisitor) {
29        (**self).visit_metrics(visitor);
30    }
31}
32
33impl<M: Metrics> Metrics for Option<M> {
34    const DESCRIPTOR: MetricGroupDescriptor = M::DESCRIPTOR;
35
36    fn visit_metrics(&self, visitor: &mut dyn MetricsVisitor) {
37        if let Some(metrics) = self {
38            metrics.visit_metrics(visitor);
39        }
40    }
41}

43/// Global instance of [`Metrics`] allowing to access contained metrics from anywhere in code.
44/// Should be used as a `static` item.
45#[derive(Debug)]
46pub struct Global<M: Metrics>(pub(crate) Lazy<M>);
47
48impl<M: Metrics + Default> Default for Global<M> {
49    fn default() -> Self {
50        Self::new()
51    }
52}

54impl<M: Metrics + Default> Global<M> {
55    /// Creates a new metrics instance.
56    pub const fn new() -> Self {
57        Self(Lazy::new(M::default))
58    }
59}

79/// Family of [`Metrics`]. Allows applying one or more labels for all contained metrics, as if each of them was enclosed in a `Family`.
80///
81/// # Examples
82///
83/// ```
84/// # use std::time::Duration;
85/// # use derive_more::Display;
86/// use vise::{
87///     Buckets, Counter, Histogram, Metrics, MetricsFamily,
88///     EncodeLabelValue, EncodeLabelSet,
89/// };
90///
91/// #[derive(Debug, Metrics)]
92/// #[metrics(prefix = "rpc_method")]
93/// struct MethodMetrics {
94///     errors: Counter,
95///     #[metrics(buckets = Buckets::LATENCIES)]
96///     latency: Histogram<Duration>,
97/// }
98///
99/// #[derive(Debug, Display, Clone, Copy, PartialEq, Eq, Hash)]
100/// #[derive(EncodeLabelValue, EncodeLabelSet)]
101/// #[metrics(label = "method")]
102/// struct Method(&'static str);
103///
104/// #[vise::register]
105/// static METRICS: MetricsFamily<Method, MethodMetrics> = MetricsFamily::new();
106///
107/// // Metrics can be accessed via indexing:
108/// METRICS[&Method("eth_call")].errors.inc();
109/// METRICS[&Method("eth_blockNumber")].latency.observe(Duration::from_millis(100));
110///
111/// // Metric instances can be stashed in a struct if needed; they have static lifetime.
112/// let call_metrics: &'static MethodMetrics = &METRICS[&Method("eth_call")];
113/// call_metrics.errors.inc_by(2);
114///
115/// let registry = vise::MetricsCollection::default().collect();
116/// let mut buffer = String::new();
117/// registry.encode(&mut buffer, vise::Format::OpenMetrics)?;
118/// assert!(buffer
119///     .lines()
120///     .any(|line| line == r#"rpc_method_errors_total{method="eth_call"} 3"#));
121/// # Ok::<_, std::fmt::Error>(())
122/// ```
123pub struct MetricsFamily<S, M: Metrics + Default>(Lazy<FamilyInner<S, M>>);
124

145impl<S, M> MetricsFamily<S, M>
146where
147    S: Clone + Eq + Hash,
148    M: Metrics + Default,
149{
150    /// Creates a new metrics family.
151    pub const fn new() -> Self {
152        Self(Lazy::new(|| FamilyInner::new(())))
153    }
154
155    /// Gets or creates metrics with the specified labels *lazily* (i.e., on first access). This is useful
156    /// if the metrics are updated conditionally and the condition is somewhat rare; in this case, indexing can
157    /// unnecessarily blow up the number of metrics in the family.
158    pub fn get_lazy(&self, labels: S) -> LazyItem<'_, S, M> {
159        LazyItem::new(&self.0, labels)
160    }
161
162    /// Returns all metrics currently present in this family together with the corresponding labels.
163    /// This is inefficient and mostly useful for testing purposes.
164    pub fn to_entries(&self) -> impl ExactSizeIterator<Item = (S, &M)> + '_ {
165        self.0.to_entries()
166    }
167}

181impl<S, M> Metrics for FamilyInner<S, M>
182where
183    S: EncodeLabelSet + Clone + Eq + Hash + Send + Sync + 'static,
184    M: Metrics + Default,
185{
186    const DESCRIPTOR: MetricGroupDescriptor = M::DESCRIPTOR;
187
188    fn visit_metrics(&self, visitor: &mut dyn MetricsVisitor) {
189        let mut grouped = LabelGroups::default();
190        for (labels, metrics) in self.to_entries() {
191            grouped.set_labels(Arc::new(labels));
192            metrics.visit_metrics(&mut grouped);
193        }
194        grouped.visit_metrics(visitor);
195    }
196}

198impl<S, M> Metrics for MetricsFamily<S, M>
199where
200    S: EncodeLabelSet + Clone + Eq + Hash + Send + Sync + 'static,
201    M: Metrics + Default,
202{
203    const DESCRIPTOR: MetricGroupDescriptor = M::DESCRIPTOR;
204
205    fn visit_metrics(&self, visitor: &mut dyn MetricsVisitor) {
206        if let Some(inner) = Lazy::get(&self.0) {
207            inner.visit_metrics(visitor);
208        }
209    }
