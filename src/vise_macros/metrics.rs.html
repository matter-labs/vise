13/// Struct-level `#[metrics(..)]` attributes.
14#[derive(Default)]
15struct MetricsAttrs {
16    cr: Option<Path>,
17    prefix: Option<LitStr>,
18}
19
20impl MetricsAttrs {
21    fn path_to_crate(&self, span: proc_macro2::Span) -> proc_macro2::TokenStream {
22        if let Some(cr) = &self.cr {
23            // Overriding the span for `cr` via `quote_spanned!` doesn't work.
24            quote!(#cr)
25        } else {
26            quote_spanned!(span=> vise)
27        }
28    }
29}
30
31impl fmt::Debug for MetricsAttrs {
32    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
33        formatter
34            .debug_struct("MetricsAttrs")
35            .finish_non_exhaustive()
36    }
37}
38
39impl ParseAttribute for MetricsAttrs {
40    fn parse(raw: &Attribute) -> syn::Result<Self> {
41        let mut attrs = Self::default();
42        raw.parse_nested_meta(|meta| {
43            if meta.path.is_ident("crate") {
44                attrs.cr = Some(meta.value()?.parse()?);
45                Ok(())
46            } else if meta.path.is_ident("prefix") {
47                attrs.prefix = Some(meta.value()?.parse()?);
48                Ok(())
49            } else {
50                Err(meta.error(
51                    "Unsupported attribute; only `prefix` and `crate` attributes are supported \
52                     (see `vise` crate docs for details)",
53                ))
54            }
55        })?;
56        Ok(attrs)
57    }
58}

99    }

102struct MetricsField {
103    attrs: MetricsFieldAttrs,
104    name: Ident,
105    ty: Type,
106    docs: String,
107}
108
109impl fmt::Debug for MetricsField {
110    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
111        formatter
112            .debug_struct("MetricsField")
113            .field("attrs", &self.attrs)
114            .field("name", &self.name)
115            .field("docs", &self.docs)
116            .finish_non_exhaustive()
117    }
118}
119
120impl MetricsField {
121    fn parse(raw: &Field) -> syn::Result<Self> {
122        let name = raw.ident.clone().ok_or_else(|| {
123            let message = "Only named fields are supported";
124            syn::Error::new_spanned(raw, message)
125        })?;
126        let ty = raw.ty.clone();
127        let attrs = metrics_attribute(&raw.attrs)?;
128
129        let doc_lines = raw.attrs.iter().filter_map(|attr| {
130            if attr.meta.path().is_ident("doc") {
131                let name_value = attr.meta.require_name_value().ok()?;
132                let Expr::Lit(doc_literal) = &name_value.value else {
133                    return None;
134                };
135                match &doc_literal.lit {
136                    Lit::Str(doc_literal) => Some(doc_literal.value()),
137                    _ => None,
138                }
139            } else {
140                None
141            }
142        });
143
144        let mut docs = String::new();
145        for line in doc_lines {
146            let line = line.trim();
147            if !line.is_empty() {
148                if !docs.is_empty() {
149                    docs.push(' ');
150                }
151                docs.push_str(line);
152            }
153        }
154        if docs.ends_with(['.', '!', '?']) {
155            // Remove the trailing punctuation since it'll be inserted automatically by the `Registry`.
156            docs.pop();
157        }
158
159        Ok(Self {
160            attrs,
161            name,
162            ty,
163            docs,
164        })
165    }
166
167    fn initialize_default(&self, cr: &proc_macro2::TokenStream) -> proc_macro2::TokenStream {
168        let name = &self.name;
169        let span = self.ty.span();
170        let mut builder = quote_spanned!(span=> #cr::MetricBuilder::new());
171        if let Some(buckets) = &self.attrs.buckets {
172            builder = quote_spanned!(span=> #builder.with_buckets(#buckets));
173        }
174        if let Some(labels) = &self.attrs.labels {
175            builder = quote_spanned!(span=> #builder.with_labels(#labels));
176        }
177
178        quote_spanned! {span=>
179            #name: #cr::BuildMetric::build(#builder)
180        }
181    }
182
183    fn visit(&self, prefix: Option<&str>) -> proc_macro2::TokenStream {
184        let name = &self.name;
185        let name_str = if let Some(prefix) = prefix {
186            format!("{prefix}_{name}")
187        } else {
188            name.to_string()
189        };
190        let docs = &self.docs;
191
192        let unit = if let Some(unit) = &self.attrs.unit {
193            quote!(::core::option::Option::Some(#unit))
194        } else {
195            quote!(::core::option::Option::None)
196        };
197
198        quote! {
199            visitor.visit_metric(
200                #name_str,
201                #docs,
202                #unit,
203                ::std::boxed::Box::new(::core::clone::Clone::clone(&self.#name)),
204            );
205        }
206    }
207
208    fn describe(
209        &self,
210        prefix: Option<&str>,
211        cr: &proc_macro2::TokenStream,
212    ) -> proc_macro2::TokenStream {
213        let name = &self.name;
214        let name_str = if let Some(prefix) = prefix {
215            format!("{prefix}_{name}")
216        } else {
217            name.to_string()
218        };
219        let docs = &self.docs;
220        let ty = &self.ty;
221        let unit = if let Some(unit) = &self.attrs.unit {
222            quote!(::core::option::Option::Some(#unit))
223        } else {
224            quote!(::core::option::Option::None)
225        };
226
227        quote! {
228            #cr::descriptors::MetricDescriptor {
229                name: #name_str,
230                field_name: ::core::stringify!(#name),
231                metric_type: <#ty as #cr::_reexports::TypedMetric>::TYPE,
232                help: #docs,
233                unit: #unit,
234            }
235        }
236    }
237}

239#[derive(Debug)]
240struct MetricsImpl {
241    attrs: MetricsAttrs,
242    name: Ident,
243    fields: Vec<MetricsField>,
244}
245
246impl MetricsImpl {
247    fn new(input: &DeriveInput) -> syn::Result<Self> {
248        ensure_no_generics(&input.generics, "Metrics")?;
249        let Data::Struct(data) = &input.data else {
250            let message = "#[derive(Metrics)] can only be placed on structs";
251            return Err(syn::Error::new_spanned(input, message));
252        };
253
254        let attrs = metrics_attribute(&input.attrs)?;
255        let name = input.ident.clone();
256        let fields = data.fields.iter().map(MetricsField::parse);
257        let fields = fields.collect::<syn::Result<_>>()?;
258        Ok(Self {
259            attrs,
260            name,
261            fields,
262        })
263    }
264
265    fn initialize(&self) -> proc_macro2::TokenStream {
266        let fields = self.fields.iter().map(|field| {
267            let cr = self.attrs.path_to_crate(field.ty.span());
268            field.initialize_default(&cr)
269        });
270
271        quote! {
272            Self {
273                #(#fields,)*
274            }
275        }
276    }
277
278    fn validate(&self) -> proc_macro2::TokenStream {
279        let prefix_assertion = self.attrs.prefix.as_ref().map(|prefix| {
280            let span = prefix.span();
281            let cr = self.attrs.path_to_crate(span);
282            quote_spanned!(span=> #cr::validation::assert_metric_prefix(#prefix);)
283        });
284        let field_assertions = self.fields.iter().map(|field| {
285            let field_ty = &field.ty;
286            let span = field_ty.span();
287            let cr = self.attrs.path_to_crate(span);
288            let type_assertion = quote_spanned! {span=>
289                { struct _AssertIsMetric where #field_ty: #cr::BuildMetric; }
290            };
291
292            let field_name = LitStr::new(&field.name.to_string(), field.name.span());
293            let span = field_name.span();
294            let cr = self.attrs.path_to_crate(span);
295            let name_assertion =
296                quote_spanned!(span=> #cr::validation::assert_metric_name(#field_name););
297            quote!(#type_assertion #name_assertion)
298        });
299        let label_assertions = self.fields.iter().filter_map(|field| {
300            let labels = field.attrs.labels.as_ref()?;
301            let span = labels.span();
302            let cr = self.attrs.path_to_crate(span);
303            Some(quote_spanned!(span=> #cr::validation::assert_label_names(&#labels);))
304        });
305
306        quote! {
307            const _: () = {
308                #prefix_assertion
309                #(#field_assertions)*
310                #(#label_assertions)*
311            };
312        }
313    }
314
315    fn implement_metrics(&self) -> proc_macro2::TokenStream {
316        let name = &self.name;
317        let cr = self.attrs.path_to_crate(name.span());
318        let prefix = self
319            .attrs
320            .prefix
321            .as_ref()
322            .map_or_else(String::new, LitStr::value);
323        let prefix = (!prefix.is_empty()).then_some(prefix.as_str());
324        let visit_fields = self.fields.iter().map(|field| field.visit(prefix));
325        let describe_fields = self.fields.iter().map(|field| field.describe(prefix, &cr));
326
327        let descriptor = quote_spanned! {name.span()=>
328            #cr::descriptors::MetricGroupDescriptor {
329                crate_name: ::core::env!("CARGO_CRATE_NAME"),
330                crate_version: ::core::env!("CARGO_PKG_VERSION"),
331                module_path: ::core::module_path!(),
332                name: ::core::stringify!(#name),
333                line: ::core::line!(),
334                metrics: &[#(#describe_fields,)*],
335            }
336        };
337
338        quote! {
339            impl #cr::Metrics for #name {
340                const DESCRIPTOR: #cr::descriptors::MetricGroupDescriptor = #descriptor;
341
342                fn visit_metrics(&self, visitor: &mut dyn #cr::MetricsVisitor) {
343                    #(#visit_fields;)*
344                }
345            }
346        }
347    }
348
349    fn derive_traits(&self) -> proc_macro2::TokenStream {
350        let name = &self.name;
351        let validation = self.validate();
352        let initialization = self.initialize();
353        let default_impl = quote! {
354            impl ::core::default::Default for #name {
355                fn default() -> Self {
356                    #initialization
357                }
358            }
359        };
360        let metrics_impl = self.implement_metrics();
361
362        quote! {
363            #validation
364            #default_impl
365            #metrics_impl
366        }
367    }
368}

370pub(crate) fn impl_metrics(input: TokenStream) -> TokenStream {
371    let input: DeriveInput = syn::parse(input).unwrap();
372    let trait_impl = match MetricsImpl::new(&input) {
373        Ok(trait_impl) => trait_impl,
374        Err(err) => return err.into_compile_error().into(),
375    };
376    trait_impl.derive_traits().into()
377}
